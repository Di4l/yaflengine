179 void TFuzzyModel::loadFromFile(std::string strFileName)
180 {
181  clear();
182 
183  nsIniFile::TIniFile file;
184  nsIniFile::TSection* section = NULL;
185  nsIniFile::TParam* param = NULL;
186  fzhndl fh_val = FL_INVALID_HANDLE;
187  std::stringstream ss_aux;
188  std::string str_aux;
189  char c_aux[512];
190  fuzzvar fv_aux;
191 
192  file.loadFile(strFileName);
193 
194  //-- Recuperamos el nombre del modelo
195  section = file[FL_SCRIPT_MODEL];
196  param = (*section)["name"];
197  name(param->value().str());
198  //-- Se crean los sets
199  section = file[FL_SCRIPT_SETS];
200  for(int i = 0; i < int(section->size()); ++i)
201  {
202  param = (*section)[i];
203  m_fsSets.add(param->name());
204  }
205 
206  //-- Para cada set, creamos los valores
207  for(int i = 0; i < int(m_fsSets.size()); ++i)
208  {
209  section = file[m_fsSets[i].name()];
210  if(!section)
211  throw TFuzzyError("loadFile", "No existe la sección",
212  __LINE__, __FILE__);
213  for(int j = 0; j < int(section->size()); ++j)
214  {
215  param = (*section)[j];
216  fh_val = m_fsSets[i].add(param->name());
217  if(FL_INVALID_HANDLE == fh_val)
218  throw TFuzzyError("loadFile",
219  "No se ha creado el valor para el set",
220  __LINE__, __FILE__);
221  m_fsSets[i][fh_val].setFunction(param->value().str());
222  }
223  }
224 
225  //-- Ahora asignamos las particularidades de cada valor en cada set
226  for(int i = 0; i < int(m_fsSets.size()); ++i)
227  {
228  for(int j = 0; j < int(m_fsSets[i].size()); ++j)
229  {
230  sprintf(c_aux, "%s_%s", m_fsSets[i].name().c_str(), m_fsSets[i][j].name().c_str());
231  str_aux = c_aux;
232 
233  section = file[str_aux];
234  if(!section)
235  throw TFuzzyError("loadFile", "No se encuentra la sección",
236  __LINE__, __FILE__);
237 
238  param = (*section)["min"];
239  if(!param)
240  throw TFuzzyError("loadFile", "No se encuentra el parametro 'min'",
241  __LINE__, __FILE__);
242  param->value() >> fv_aux;
243  m_fsSets[i][j].min(fv_aux);
244 
245  param = (*section)["max"];
246  if(!param)
247  throw TFuzzyError("loadFile", "No se encuentra el parametro 'max'",
248  __LINE__, __FILE__);
249  param->value() >> fv_aux;
250  m_fsSets[i][j].max(fv_aux);
251 
252  param = (*section)["count"];
253  if(!param)
254  throw TFuzzyError("loadFile", "No se encuentra el parametro 'count'",
255  __LINE__, __FILE__);
256  param->value() >> fv_aux;
257  m_fsSets[i][j].size(size_t(fv_aux));
258 
259  if(m_fsSets[i][j].size() > 0)
260  {
261  char c_par[10];
262  for(size_t k = 0; k < m_fsSets[i][j].size(); ++k)
263  {
264  sprintf(c_par, "param_%04u", k);
265  param = (*section)[c_par];
266  param->value() >> m_fsSets[i][j][k];
267  }
268  }
269  }
270  }
271 
272  //-- Finalmente se cargan las reglas
273  section = file[FL_SCRIPT_RULES];
274  if(!section)
275  throw TFuzzyError("loadFile", "No se encuentran reglas para este modelo",
276  __LINE__, __FILE__);
277 
278  for(size_t i = 0; i < section->size(); ++i)
279  {
280  m_frRules.add((*section)[i]->value().str());
281  }
282 
283  file.close();
284 }

290 void TFuzzyModel::saveToFile(std::string strFileName, bool bAddComments)
291 {
292  char c_aux[32];
293  std::string str_aux;
294  std::stringstream ss_aux;
295  nsIniFile::TIniFile file;
296  nsIniFile::TSection* section = NULL;
297  nsIniFile::TParam* param = NULL;
298 
299  //-- Se añade la cabecera al archivo de configuración del modelo
300  if(bAddComments)
301  configurationHint(file.header());
302  //-- Se guarda el nombre del modelo
303  section = file.add(FL_SCRIPT_MODEL);
304  if(bAddComments)
305  section->comment() = m_strDescription;
306  param = section->add("name");
307  param->value() << m_strName;
308 
309  //-- Se guarda la lista de sets
310  section = file.add(FL_SCRIPT_SETS);
311  if(bAddComments)
312  section->comment() = "Lista los sets que componen el modelo";
313  for(int i = 0; i < int(m_fsSets.size()); ++i)
314  {
315  param = section->add(m_fsSets[i].name());
316  param->value() << m_fsSets[i].size();
317  }
318 
319  //-- Se guarda las características de cada set
320  for(int i = 0; i < int(m_fsSets.size()); ++i)
321  {
322  section = file.add(m_fsSets[i].name());
323  if(bAddComments)
324  section->comment() = "Definición del conjunto " + section->name();
325  for(int j = 0; j < int(m_fsSets[i].size()); ++j)
326  {
327  param = section->add(m_fsSets[i][j].name());
328  param->value() << m_fsSets[i][j].function()->name();
329  }
330  }
331 
332  //-- Se guarda los valores de cada set
333  for(int i = 0; i < int(m_fsSets.size()); ++i)
334  {
335  for(int j = 0; j < int(m_fsSets[i].size()); ++j)
336  {
337  str_aux = m_fsSets[i].name() + std::string("_") + m_fsSets[i][j].name();
338  section = file.add(str_aux);
339  if(bAddComments)
340  section->comment() = "Definición del valor " + m_fsSets[i].name() + "." + m_fsSets[i][j].name();
341  param = section->add("min");
342  param->value() << m_fsSets[i][j].min();
343  if(bAddComments)
344  param->comment() = "Valor mínimo que toma";
345  param = section->add("max");
346  param->value() << m_fsSets[i][j].max();
347  if(bAddComments)
348  param->comment() = "Valor máximo que toma";
349  param = section->add("count");
350  param->value() << m_fsSets[i][j].size();
351  if(bAddComments)
352  param->comment() = "Número de parámetros que necesita la función";
353 
354  if(m_fsSets[i][j].size() > 0)
355  {
356  char c_par[11];
357  for(size_t k = 0; k < m_fsSets[i][j].size(); ++k)
358  {
359  sprintf(c_par, "param_%04u", k);
360  param = section->add(c_par);
361  param->value() << m_fsSets[i][j][k];
362  }
363  }
364  }
365  }
366 
367  //-- Se guarda las reglas
368  section = file.add(FL_SCRIPT_RULES);
369  if(bAddComments)
370  section->comment() = "Establece la relación que hay entre los diferentes conjuntos del modelo";
371  for(size_t i = 0; i < m_frRules.size(); ++i)
372  {
373  sprintf(c_aux, "rule_%03u", i + 1);
374 
375  param = section->add(c_aux);
376  param->value() << m_frRules[i].String;
377  }
378 
379  file.saveFile(strFileName);
380  file.close();
381 }
