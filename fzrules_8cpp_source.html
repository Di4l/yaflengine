137 SFuzzyRule* TFuzzyRules::parseRule(std::string strRule)
138 {
139  toLower(strRule);
140 
141  if(strRule.find("if ") == std::string::npos)
142  {
143  throw TFuzzyError("parseRule", "No se encuentra la clausula 'if'",
144  __LINE__, __FILE__);
145  }
146  if(strRule.find(" then ") == std::string::npos)
147  {
148  throw TFuzzyError("parseRule", "No se encuentra la clausula 'then'",
149  __LINE__, __FILE__);
150  }
151 
152  std::string str_rule = strRule;
153  SFuzzyRule* fr_rule = NULL;
154  size_t pos = strRule.find("if ");
155  SRuleAtom *f_link = NULL, *f_first = NULL;
156  std::string str_cmd, str_val, str_set;
157  std::string str_mods;
158 
159  strRule = strRule.substr(pos + 3);
160  trim(strRule);
161 
162  pos = strRule.find(" and ") != std::string::npos
163  ? strRule.find(" and ") : strRule.find(" then ");
164 
165  try
166  {
167  while(!strRule.empty())
168  {
169  //-- Obtiene el 'comando' (set.[mod.]value) y prepara strRule para
170  // la siguiente pasada a este while
171  str_cmd = strRule.substr(0, pos);
172  trim(str_cmd);
173  if(pos == strRule.size())
174  pos -= 5;
175  strRule = strRule.substr(pos + 5);
176  trim(strRule);
177 
178  //-- Al menos debe haber un punto (set.value)... si no hay ninguno,
179  // error en la regla
180  pos = str_cmd.find(".");
181  if(pos == std::string::npos)
182  throw TFuzzyError("parseRule", "No se ha encontrado el operador '.'",
183  __LINE__, __FILE__);
184 
185  //-- Obtiene el set
186  str_set = str_cmd.substr(0, pos);
187  if(m_fsSets[str_set] == invalidObject())
188  {
189  std::string str_why = "No existe el set " + str_set;
190  throw TFuzzyError("parseRule", str_why.c_str(), __LINE__, __FILE__);
191  }
192 
193  //-- Obtiene el valor y busca un modificador ([mod.]value)
194  str_val = str_cmd.substr(pos + 1);
195  str_mods = splitModsValue(str_val);
196  if(m_fsSets[str_set][str_val] == invalidObject())
197  {
198  std::string str_why = "No existe el valor " + str_val + "en el set " + str_set;
199  throw TFuzzyError("parseRule", str_why.c_str(), __LINE__, __FILE__);
200  }
201 
202  if(f_first)
203  {
204  f_link->Next = new SRuleAtom(&m_fsSets[str_set][str_val]);
205  f_link = f_link->Next;
206  }
207  else
208  {
209  fr_rule = new SFuzzyRule(&m_fsSets[str_set][str_val]);
210  fr_rule->String = str_rule;
211  f_first = fr_rule->Links;
212  f_link = f_first;
213  }
214  //-- Asigna los modificadores al valor
215  setModifiers(f_link, str_mods);
216 
217  pos = strRule.find(" and ") != std::string::npos
218  ? strRule.find(" and ") : strRule.find(" then ");
219 
220  if(pos == std::string::npos)
221  pos = strRule.size();
222  }
223  }
224  catch(TFuzzyError& e)
225  {
226  if(fr_rule)
227  delete fr_rule;
228  fr_rule = NULL;
229 
230  throw e;
231  }
232  catch(...)
233  {
234  if(fr_rule)
235  delete fr_rule;
236  fr_rule = NULL;
237 
238  throw TFuzzyError("parseRule", "Error sin especificar", __LINE__, __FILE__);
239  }
240 
241  return fr_rule;
242 }
