120 fuzzvar TExecData::ruleResult(SRuleAtom* raAtom)
121 {
122  fuzzvar fv_ret = FZ_INVALID_RESULT;
123 
124  if(raAtom)
125  {
126  fuzzmap::iterator fm_val = m_fmResults.find(raAtom->Value->handle());
127  if(fm_val != m_fmResults.end())
128  {
129  fv_ret = fm_val->second;
130 
131  for(size_t i = 0; i < raAtom->Modifiers.size(); ++i)
132  {
133  switch(raAtom->Modifiers[i])
134  {
135  case vmVery:
136  fv_ret = pow(double(fv_ret), double(2.0));
137  break;
138 
139  case vmSlightly:
140  fv_ret = sqrt(fv_ret);
141  break;
142 
143  case vmNot:
144  fv_ret = 1.0 - fv_ret;
145  break;
146 
147  case vmNone:
148  default:
149  break;
150  }
151  }
152  }
153  else
154  {
155  std::string str_why = std::string("No result stored for the value '")
156  + raAtom->Value->parent()->name()
157  + std::string(".")
158  + raAtom->Value->name()
159  + std::string("'.");
160  throw TFuzzyError("ruleResult", str_why.c_str(), __LINE__, __FILE__);
161  }
162  }
163  else
164  {
165  throw TFuzzyError("ruleResult", "No rule to apply modifiers found.", __LINE__, __FILE__);
166  }
167 
168  return fv_ret;
169 }

172 void TExecData::setLimits()
173 {
174  fuzzmap::iterator fm_val;
175  fuzzmap::iterator fm_lim;
176  fuzzvar fv_aux;
177  fuzzvar fv_res;
178  SRuleAtom* ra_rule = NULL;
179  fuzzrullist::iterator iter = m_rlRules.begin();
180 
181  resetLimits();
182  while(iter != m_rlRules.end())
183  {
184  fv_aux = 1.0;
185  ra_rule = *iter;
186 
187  //-- Apply the 'ands' of the rule (keep the lowest value)
188  while(ra_rule && ra_rule->Next)
189  {
190  fv_res = ruleResult(ra_rule);
191  fv_aux = fv_res < fv_aux ? fv_res : fv_aux;
192  ra_rule = ra_rule->Next;
193  }
194 
195  //-- Apply the 'or' of the rules (keep the highest)
196  fm_lim = m_fmLimits.find(ra_rule->Value->handle());
197  fm_lim->second = fv_aux > fm_lim->second ? fv_aux : fm_lim->second;
198 
199  ++iter;
200  }
201 }

204 fuzzlist& TExecData::fuzzyfy(fuzzlist& flOutput)
205 {
206  TFuzzyVal* fs_val = NULL;
207  int i_index;
208  fuzzvar x_tmp;
209  fuzzvar fv_tmp;
210  fuzzvar fv_val;
211 
212  fuzzmap::iterator fl_lim;
213 
214  flOutput.clear();
215  for(int i = 0; i < FL_CRV_COUNT + 1; ++i)
216  {
217  i_index = 2 * i;
218  x_tmp = m_fsSet->min() + i * (m_fsSet->max() - m_fsSet->min()) / FL_CRV_COUNT;
219  flOutput.push_back(x_tmp);
220  flOutput.push_back(0.0);
221 
222  for(int j = 0; j < int(m_fsSet->size()); ++j)
223  {
224  fs_val = &(fuzzySet()[j]);
225  fl_lim = m_fmLimits.find(fs_val->handle());
226  fv_val = fs_val->execute(x_tmp);
227  fv_tmp = 0.0;
228 
229  fv_tmp = fv_val < fl_lim->second ? fv_val : fl_lim->second;
230 
231  flOutput[i_index + 1] = flOutput[i_index + 1] > fv_tmp
232  ? flOutput[i_index + 1] : fv_tmp;
233  }
234  }
235 
236  return flOutput;
237 }

240 fuzzvar TExecData::defuzzyfy(fuzzlist& flData)
241 {
242  if(0 == flData.size())
243  throw TFuzzyError("defuzzyfy",
244  "No hay valores con los que calcular", __LINE__, __FILE__);
245 
246  fuzzvar left = 0.0;
247  fuzzvar right = 0.0;
248  fuzzvar dx;
249  int i_left = 0;
250  int i_right = flData.size() - 2;
251  bool b_left = true;
252 
253  while(i_left + 2 < i_right)
254  {
255  if(b_left)
256  {
257  dx = flData[i_left + 2] - flData[i_left];
258  left += (dx * (flData[i_left + 3] + flData[i_left + 1]) / 2.0);
259  i_left += 2;
260  }
261  else
262  {
263  dx = flData[i_right] - flData[i_right - 2];
264  right += (dx * (flData[i_right + 1] + flData[i_right - 1]) / 2.0);
265  i_right -= 2;
266  }
267  b_left = left < right ? true : false;
268  }
269 
270  // For a first approximation...the error is +- dx / 2.0
271  return (flData[i_left] + flData[i_right]) / 2.0;
272 }

309 fuzzvar TExecData::execute(fuzzvar x)
310 {
311  if(m_fvResult != x)
312  {
313  TFuzzyVal* fz_val = NULL;
314  for(int i = 0; i < int(m_fsSet->size()); ++i)
315  {
316  fz_val = &(fuzzySet()[i]);
317  m_fmResults[fz_val->handle()] = fz_val->execute(x);
318  }
319

321  }
322  //-- We flag ourselves as set, as we have just calculated the outcome

324

329 fuzzvar TExecData::calculate()
330 {
331  //-- If we have no Inputs it means we are a pure input and need no
332  // further calculation. m_fvResult already has the value for this set.
333  if(m_emInputs.empty())
334  return m_fvResult;
335 
336  //-- If we already have been calculated, do not do it again.
337  switch(m_esStatus)
338  {
339  case esSet:
340  return m_fvResult;
341 
342  case esUnset:
343  default:
344  break;
345  }
346 
347  //-- If we have inputs we depend upon, calculate them first
348  fuzzexemap::iterator iter = m_emInputs.begin();
349 
350  while(iter != m_emInputs.end())
351  {
352  iter->second->calculate();
353  ++iter;
354  }
355 
356  //-- Once all my inputs have been set to their values, we can proceed
357  // to calculate my outcome.
358  setLimits();
359  fuzzlist fl_outcurve;
360  return execute(defuzzyfy(fuzzyfy(fl_outcurve)));
361 }

411 void TFuzzyExec::setProcessChain()
412 {
413  if(!m_fmModel)
414  return;
415 
416  createData();
417 
418  SRuleAtom* fl_link = NULL;
419  SRuleAtom* fl_out = NULL;
420  fzhndl fh_id, fh_out;
421 
422  //-- First step: Fill the input and output trees from
423  // the rules.
424  for(size_t i = 0; i < m_fmModel->rules().size(); ++i)
425  {
426  fl_link = m_fmModel->rules()[i].Links;
427  fl_out = fl_link;
428 
429  while(fl_out->Next)
430  fl_out = fl_out->Next;
431 
432  fh_out = fl_out->Value->parent()->handle();
433 
434  m_mExecData[fh_out]->rules().push_back(fl_link);
435 
436  while(fl_link->Next)
437  {
438  fh_id = fl_link->Value->parent()->handle();
439 
440  //-- Add the inputs as siblings of the output of this rule.
441  m_mExecData[fh_out]->add(fh_id, m_mExecData[fh_id]);
442 
443  fl_link = fl_link->Next;
444  }
445  }
446 }

512 void TFuzzyExec::calculate()
513 {
514  if(!m_fmModel)
515  return;
516  //-- All sets should have their input set by now
517 
518  //-- 1st: Starting from the output tree, travel down calculating its output
519  fuzzexemap::iterator iter = m_mExecData.begin();
520 
521  //-- Assign all output sets the "unset" status so that they are all
522  // recalculated
523  while(iter != m_mExecData.end())
524  {
525  if(!iter->second->inputs().empty())
526  iter->second->unset();
527  ++iter;
528  }
529 
530  iter = m_mExecData.begin();
531  while(iter != m_mExecData.end())
532  {
533  iter->second->calculate();
534  ++iter;
535  }
536 }
