117 std::string& TFuzzyBase::trim(std::string& strText)
118 {
119  size_t pos = strText.find_first_not_of(" ");
120  if(pos != std::string::npos)
121  strText = strText.substr(pos);
122 
123  pos = strText.size() - 1;
124  while(strText[pos] == ' ' && pos != 0)
125  --pos;
126  if(pos >= 0)
127  strText = strText.substr(0, pos + 1);
128 
129  return strText;
130 }
