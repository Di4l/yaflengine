100 fuzzvar TFuzzyVal::calcMaxFunc()
101 {
102  fuzzvar x = min();
103  fuzzvar y = 0.0;
104  fuzzvar dy = 2.0;
105  fuzzvar Dx = (max() - min()) / 2;
106  fuzzvar de2 = 1.0E-6;
107 
108  while(dy > de2)
109  {
110  x += Dx;
111 
112  dy = y;
113  y = exeFunction(x);
114 
115  if(y < dy)
116  Dx /= -2.0;
117 
118  dy = pow(y - dy, 2.0);
119 
120  if(x > max())
121  {
122  x = max();
123  dy = 0.0;
124  }
125  else if(x < min())
126  {
127  x = min();
128  dy = 0.0;
129  }
130  }
131 
132  return x;
133 }
