94 /*
95  * Las funciones de grado de verdad dentro del sistema pueden accederse a
96  * través de su descriptor o de su nombre. Esto último requiere que el nombre
97  * de cada función dentro del sistema sea único.
98  *
99  * Con esta filosofía, pueden darse los siguientes dos casos a tener en cuenta:
100  *
101  * \li Que dos o más funciones con nombre diferente apunten a la misma función.
102  * Esto está permitido. Puede verse este caso como que los diferentes nombres
103  * son alias de una misma función.
104  * \li Que dos o más funciones con el mismo nombre apunten a diferentes
105  * funciones. Esto NO está permitido. En el sistema queda registrada la primera
106  * función que se haya registrado. Las demás serán ignoradas.
107  *
108  * \return false si el Nombre de las funciones coincide. true en cualquier otro
109  * caso.
110  * \sa TStdFuzzyFunctions, TStdFuzzyFunctions::add()
111  */
112 bool SFuzzyFunction::operator !=(SFuzzyFunction& cmp)
113 {
114  return name() != cmp.name();
115 }

242 fuzzvar TStdFuzzyFunctions::gaussBell(fuzzlist& params, fuzzvar x)
243 {
244  if(params.size() < 2)
245  throw TFuzzyError("gausBell", "Número de parámetros incorrecto",
246  __LINE__, __FILE__);
247 
248  fuzzvar width = (abs(params[0]) + abs(params[1])) / 2.0;
249  fuzzvar center = (params[0] + params[1]) / 2.0;
250  fuzzvar attenu = -log(0.001) / width / width;
251  fuzzvar X = (x - center) * (x - center);
252 
253  return fuzzvar(exp(-attenu * X));
254 }

257 fuzzvar TStdFuzzyFunctions::sCurve(fuzzlist& params, fuzzvar x)
258 {
259  if(params.size() < 2)
260  throw TFuzzyError("sCurve", "Número de parámetros incorrecto",
261  __LINE__, __FILE__);
262 
263  if(x < params[0])
264  return 0.0;
265  if(x > params[1])
266  return 1.0;
267 
268  fuzzvar rtn, A, center;
269  A = 2.0 / (params[0] - params[1]) / (params[0] - params[1]);
270  center = (params[0] + params[1]) / 2.0;
271 
272  rtn = x > center
273  ? 1.0 - A * (x - params[1]) * (x - params[1])
274  : A * (x - params[0]) * (x - params[0]);
275 
276  return rtn;
277 }

286 fuzzvar TStdFuzzyFunctions::triangle(fuzzlist& params, fuzzvar x)
287 {
288  size_t sz_cnt = params.size();
289 
290  if(sz_cnt < 2)
291  throw TFuzzyError("triangle", "Número de parámetros incorrecto",
292  __LINE__, __FILE__);
293 
294  fuzzvar mid = sz_cnt >= 3 ? params[2] : (params[0] + params[1]) / 2.0;
295  fuzzvar b = 1.0 / (mid - (x < mid ? params[0] : params[1]));
296  fuzzvar a = 1.0 - b * mid;
297 
298  return fuzzvar((x > params[0] && x < params[1]) ? a + b * x : 0.0);
299 }

308 fuzzvar TStdFuzzyFunctions::interpolate(fuzzlist& params, fuzzvar x)
309 {
310  // params viene en pares... x,y, x,y, x,y...
311  size_t sz_cnt = params.size();
312 
313  if(sz_cnt < 2)
314  throw TFuzzyError("interpolate", "Número de parámetros incorrecto",
315  __LINE__, __FILE__);
316  if(sz_cnt % 2)
317  throw TFuzzyError("interpolate", "Número de parámetros debe ser par",
318  __LINE__, __FILE__);
319 
320  fuzzvar rtn;
321  size_t i_index = 0;
322  while(i_index < sz_cnt && params[2 * i_index] < x)
323  ++i_index;
324 
325  if(i_index >= sz_cnt)
326  rtn = params[2 * i_index - 1];
327  else
328  {
329  fuzzvar Dy = params[2 * i_index + 1] - params[2 * i_index - 1];
330  fuzzvar Dx = params[2 * i_index] - params[2 * (i_index - 1)];
331  fuzzvar dx = x - params[2 * (i_index - 1)];
332  rtn = dx * Dy / Dx + params[2 * i_index - 1];
333  }
334  return rtn;
335 }

343 fzhndl TStdFuzzyFunctions::add(SFuzzyFunction* fFunc)
344 {
345  if(!fFunc)
346  return INVALID_FUNC;
347  if(fFunc->name().empty())
348  throw TFuzzyError("add", "Nombre de función vacío", __LINE__, __FILE__);
349  if(fFunc->Function == NULL)
350  throw TFuzzyError("add", "Puntero a función NULL", __LINE__, __FILE__);
351 
352  std::map<fzhndl, SFuzzyFunction*>::iterator iter = m_mFunctions.begin();
353 
354  while(iter != m_mFunctions.end() && fFunc != iter->second && *fFunc != *(iter->second))
355  ++iter;
356 
357  if(iter == m_mFunctions.end())
358  {
359  m_mFunctions[fFunc->handle()] = fFunc;
360  return fFunc->handle();
361  }
362  else if(iter->second == fFunc || iter->second->name() == fFunc->name())
363  {
364  return iter->first;
365  }
366 
367  return INVALID_FUNC;
368 }
