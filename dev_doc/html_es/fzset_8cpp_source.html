323 fzhndl TFuzzySets::add(std::string strName)
324 {
325  TFuzzySet* fs_new = NULL;
326  std::map<fzhndl, TFuzzySet*>::iterator iter = m_mSets.begin();
327 
328  while(iter != m_mSets.end())
329  {
330  if(iter->second->name() == strName)
331  break;
332  ++iter;
333  }
334 
335  if(iter == m_mSets.end())
336  {
337  fs_new = new TFuzzySet(strName);
338  fs_new->parent() = this;
339  m_mSets[fs_new->handle()] = fs_new;
340  }
341 
342  return iter == m_mSets.end() ?
343  (fs_new ? fs_new->handle() : FL_INVALID_HANDLE) :
344  iter->first;
345 }
